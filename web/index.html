Why Choose Raven?

⚡

Native Compilation

Compiled to native machine code through Cranelift. One static binary, no VM and no interpreter.

🧩

A Real Type System

Generics with trait bounds, traits for polymorphism, and sum types checked by an exhaustive match.

🛡️

Safe by Default

A tracing garbage collector handles memory, and Result and Option replace null. No manual frees, no null pointers.

🧵

Concurrency

Lightweight goroutines with spawn, and channels for passing values between them.

🔌

C FFI

Call into native libraries with extern "C": numeric types, raw pointers, callbacks, and small structs by value.

🛠️

Metaprogramming & Tooling

@derive and macros, compile-time and runtime reflection, GitHub-direct packages with rvpm, one formatter, and a VS Code extension.

Code Examples

// A map literal needs the collections module
import std/collections

// Inferred types, no semicolons
let name = "Raven"
let version = 2

// Collections and literals
let numbers: List<Int> = [1, 2, 3]
let scores = ["ada": 95, "alan": 88]

// Range-based for and string interpolation
for n in 0..3 {
    print("${name} v${version}: ${n}")
}
// A trait and an implementation
trait Shape {
    fun area(self) -> Float
}

struct Circle { radius: Float }

impl Shape for Circle {
    fun area(self) -> Float = 3.14159 * self.radius * self.radius
}

// A generic function taking a closure
fun max_by<T>(a: T, b: T, key: fun(T) -> Int) -> T {
    if key(a) >= key(b) { return a }
    return b
}
// An enum with payload variants
enum Shape {
    Circle(Float),
    Rect(Float, Float),
}

// Construct with EnumName.Variant, match with bare names
fun area(s: Shape) -> Float {
    return match s {
        Circle(r) -> 3.14159 * r * r,
        Rect(w, h) -> w * h,
    }
}

// Errors live in the type, propagated with ?
fun half(n: Int) -> Result<Int, String> {
    if n % 2 != 0 { return Err("odd") }
    return Ok(n / 2)
}
// Goroutines and channels
import std/sync { channel }

fun main() {
    let ch = channel()

    // A goroutine sends 1 through 5
    spawn(fun() -> Unit {
        let i = 1
        while i <= 5 {
            ch.send(i)
            i = i + 1
        }
    })

    let sum = 0
    let n = 0
    while n < 5 {
        sum = sum + ch.recv()
        n = n + 1
    }
    print(sum)  // 15
}
